	return del('dist/**/*', { force: true }) // –£–¥–∞–ª—è–µ–º –≤—Å—ë —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ –ø–∞–ø–∫–∏ "dist/"
}

    function startwatch() {
	// –í—ã–±–∏—Ä–∞–µ–º –≤—Å–µ —Ñ–∞–π–ª—ã JS –≤ –ø—Ä–æ–µ–∫—Ç–µ, –∞ –∑–∞—Ç–µ–º –∏—Å–∫–ª—é—á–∏–º —Å —Å—É—Ñ—Ñ–∏–∫—Å–æ–º .min.js
	watch(['app/**/*.js', '!app/**/*.min.js'], scripts);
	
	// –ú–æ–Ω–∏—Ç–æ—Ä–∏–º —Ñ–∞–π–ª—ã –ø—Ä–µ–ø—Ä–æ—Ü–µ—Å—Å–æ—Ä–∞ –Ω–∞ –∏–∑–º–µ–Ω–µ–Ω–∏—è
	watch('app/**/' + preprocessor + '/**/*', styles);
 
	// –ú–æ–Ω–∏—Ç–æ—Ä–∏–º —Ñ–∞–π–ª—ã HTML –Ω–∞ –∏–∑–º–µ–Ω–µ–Ω–∏—è
	watch('app/**/*.html').on('change', browserSync.reload);
 
	// –ú–æ–Ω–∏—Ç–æ—Ä–∏–º –ø–∞–ø–∫—É-–∏—Å—Ç–æ—á–Ω–∏–∫ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π –∏ –≤—ã–ø–æ–ª–Ω—è–µ–º images(), –µ—Å–ª–∏ –µ—Å—Ç—å –∏–∑–º–µ–Ω–µ–Ω–∏—è
	watch('app/images/src/**/*', images);
}

    // –≠–∫—Å–ø–æ—Ä—Ç–∏—Ä—É–µ–º —Ñ—É–Ω–∫—Ü–∏—é browsersync() –∫–∞–∫ —Ç–∞—Å–∫ browsersync. –ó–Ω–∞—á–µ–Ω–∏–µ –ø–æ—Å–ª–µ –∑–Ω–∞–∫–∞ = —ç—Ç–æ –∏–º–µ—é—â–∞—è—Å—è —Ñ—É–Ω–∫—Ü–∏—è.
