Выборка и получение контекста

var canvas = document.getElementById('canvas'); var ctx = canvas.getContext('2d'); canvas.width = 200; canvas.height = 200; //canvas.width = window.innerWidth; //canvas.height = window.innerHeight;

Общие свойства

ctx.fillStyle = 'yellowgreen'; ctx.strokeStyle = 'black'; ctx.lineWidth = 4;
